  3 this.msgpack.rpc || (function(globalScope) {
  4 
  5 globalScope.msgpack.rpc = {
  6     /**
  7      * MessagePack RPC Client<br>
  8      * <a href="http://wiki.msgpack.org/display/MSGPACK/RPC+specification">
  9      * http://wiki.msgpack.org/display/MSGPACK/RPC+specification
 10      * </a>
 11      * @class
 12      * @param {string} uri uri of websocket
 13      * @param {Function} [event_callback]
 14      * called when received event<br>
 15      * @return {Object}
 16      * MessagePack RPC Client Instance or undefined if error occurs
 17      * @see #event:event_callback
 18      * @example
 19      * (function() {
 20      *   function event_callback(e) {
 21      *     console.log(JSON.stringify(e));   
 22      *   }
 23      * 
 24      *   function responce_callback_1(r) {
 25      *     console.log(JSON.stringify(r));
 26      *   }
 27      * 
 28      *   function responce_callback_2(r) {
 29      *     console.log(JSON.stringify(r));
 30      *   }
 31      * 
 32      *   var client = new msgpack.rpc.client('ws://host:port/msgpack-rpc',
 33      *                                       event_callback);
 34      *   // non-block
 35      *   client.call_async({'method': 'foo',
 36      *                      'params': [-1, "string", {'key': 'val'}],
 37      *                      'callback': response_callback_1,
 38      *                      'timeout': 10000});
 39      *   client.call_async({'method': 'bar',
 40      *                      'params': [-1, "string", {'key': 'val'}],
 41      *                      'callback': response_callback_2,
 42      *                      'timeout': 10000});
 43      * })();
 44      */
 45     client: msgpackclient
 46 };
 47     function msgpackclient(uri, event_callback) {
 48         var sock, connected = false, msgid = -1;
 49         var requests = {}, that = {}, unpacker = new msgpack.unpacker();
 50 
 51         function send_request(id, rpc) {
 52             var frame, pack, ui8v, timeout = 30000;
 53 
 54             function send() {
 55                 if (sock.bufferedAmount == 0) {
 56                     timeout = rpc.timeout || 30000;
 57                     requests[id].tid = setTimeout(function() {
 58                                                           timeout_request(id);
 59                                                       }, timeout);
 60                     console.log('set', id, requests[id].tid);
 61                     sock.send(ui8v.buffer);
 62                 } else {
 63                     setTimeout(send, 50);
 64                 }
 65             }
 66 
 67             requests[id] = {};
 68             if (rpc.callback && typeof rpc.callback === 'function') {
 69                 requests[id].callback = rpc.callback;
 70             }
 71             frame = [0, id, rpc.method];
 72             frame.push(rpc.params);
 73             pack = msgpack.pack(frame);
 74             ui8v = new Uint8Array(pack);
 75             send();
 76         }
 77         function timeout_request(id) {
 78             if (typeof requests[id] !== 'undefined') {
 79                 requests[id].callback({'error': 'timeout', 'result': null});
 80                 delete requests[id];
 81             }
 82         }
 83         function recv_response(e) {
 84             var pack = [], r, event = {};
 85             var ui8v = new Uint8Array(e.data);
 86 
 87             for (var i = 0; i < ui8v.length; i++) {
 88                 pack[i] = ui8v[i];
 89             }
 90             unpacker.feed(pack);
 91             while ((r = unpacker.unpack()) !== undefined) {
 92                 if (r[0] == 1 && typeof requests[r[1]] !== 'undefined') {
 93                     console.log('clear', r[1], requests[r[1]].tid);
 94                     clearTimeout(requests[r[1]].tid);
 95                     requests[r[1]].callback({'error': r[2], 'result': r[3]});
 96                     delete requests[r[1]];
 97                 }
 98                 if (r[0] == 2) {
 99                     event.type = 'message';
100                     event.method = r[1];
101                     event.params = r[2];
102                     recv_event(event);
103                 }
104             }
105         }
106         function recv_event(e) {
107             if (typeof that.event_callback === 'function') {
108                 that.event_callback(e);
109             }
110         }
111 
112         /**
113          * do RPC async
114          * @memberOf globalScope.msgpack.rpc.client.prototype
115          * @param {Hash} rpc
116          * @param {String} rpc.method method name for RPC
117          * @param {Array} rpc.params params for RPC
118          * @param {Function} [rpc.callback]
119          * called this function when received response<br>
120          * @param {Number} [rpc.timeout]
121          * timeout time[ms], default 30000 ms
122          * @see #event:response_callback
123          */
124          that.call_async = function(rpc) {
125             msgid = (msgid == 0x0ffffffff) ? 0 : msgid + 1;
126             if (!connected || typeof requests[msgid] !== 'undefined') {
127                 setTimeout(function() {
128                                that.call_async(rpc);
129                            }, 50);
130                 return;
131             }
132             send_request(msgid, rpc);
133         };
134 
135         // initialize
136         try {
137             sock = new WebSocket(uri);   
138         } catch (x) {
139             return undefined;
140         }
141         sock.binaryType = 'arraybuffer';
142         sock.onopen = function() {
143             connected = true;
144         };
145         sock.onclose = sock.onerror = recv_event;
146         sock.onmessage = recv_response;
147         that.event_callback = event_callback;
148         return that;
149         /**
150          * fire when received event message from a server
151          * @name globalScope.msgpack.rpc.client#event_callback
152          * @event
153          * @param {Hash} e
154          * @param {String} e.type type of event
155          * [close, error, message]
156          * @param {String} [e.method] method name
157          * @param {Array} [e.params] arguments array that a server defines
158          */
159         /**
160          * fire when received response
161          * @name globalScope.msgpack.rpc.client#responce_callback
162          * @event
163          * @param {Hash} r
164          * @param {String} r.error
165          * error message from a server
166          * @param {Object that you define} r.result
167          * response object from a server
168          */
169     }
170 })(this);
