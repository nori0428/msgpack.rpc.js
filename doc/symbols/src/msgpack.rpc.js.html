  2 
  3 this.msgpack.rpc || (function(globalScope) {
  4 
  5 globalScope.msgpack.rpc = {
  6     /**
  7      * MessagePack RPC Client<br>
  8      * <a href="http://wiki.msgpack.org/display/MSGPACK/RPC+specification">
  9      * http://wiki.msgpack.org/display/MSGPACK/RPC+specification
 10      * </a>
 11      * @class
 12      * @param {string} uri uri of websocket
 13      * @param {Hash} [callbacks]
 14      * fire when receive event or notify
 15      * @param {String} [callbacks.event]
 16      * fire when receive websocket event
 17      * @param {String} [callbacks.notify]
 18      * fire when receive MessagePack notification
 19      * @return {Object}
 20      * MessagePack RPC Client Instance or undefined if error occurs
 21      * @see #event:event_callback
 22      * @see #event:notify_callback
 23      * @example
 24      * (function() {
 25      *   function notify_callback(e) {
 26      *     console.log(JSON.stringify(e));   
 27      *   }
 28      * 
 29      *   function event_callback(e) {
 30      *     console.log(JSON.stringify(e));   
 31      *   }
 32      * 
 33      *   function responce_callback_1(r) {
 34      *     console.log(JSON.stringify(r));
 35      *   }
 36      * 
 37      *   function responce_callback_2(r) {
 38      *     console.log(JSON.stringify(r));
 39      *   }
 40      * 
 41      *   var client = new msgpack.rpc.client('ws://host:port/msgpack-rpc',
 42      *                                       {'notify': notify_callback,
 43      *                                        'event': event_callback});
 44      *   // non-block
 45      *   client.call_async({'method': 'foo',
 46      *                      'params': [-1, "string", {'key': 'val'}],
 47      *                      'callback': response_callback_1,
 48      *                      'timeout': 10000});
 49      *   client.call_async({'method': 'bar',
 50      *                      'params': [-1, "string", {'key': 'val'}],
 51      *                      'callback': response_callback_2,
 52      *                      'timeout': 10000});
 53      * })();
 54      */
 55     client: msgpackclient
 56 };
 57     function msgpackclient(uri, callbacks) {
 58         var sock, connected = false, msgid = -1;
 59         var requests = {}, that = {}, unpacker = new msgpack.unpacker();
 60 
 61         function send_request(id, rpc) {
 62             var frame, pack, ui8v, timeout = 30000;
 63 
 64             function send() {
 65                 if (sock.bufferedAmount == 0) {
 66                     timeout = rpc.timeout || 30000;
 67                     requests[id].tid = setTimeout(function() {
 68                                                           timeout_request(id);
 69                                                       }, timeout);
 70                     sock.send(ui8v.buffer);
 71                 } else {
 72                     setTimeout(send, 50);
 73                 }
 74             }
 75 
 76             requests[id] = {};
 77             if (rpc.callback && typeof rpc.callback === 'function') {
 78                 requests[id].callback = rpc.callback;
 79             }
 80             frame = [0, id, rpc.method];
 81             frame.push(rpc.params);
 82             pack = msgpack.pack(frame);
 83             ui8v = new Uint8Array(pack);
 84             send();
 85         }
 86         function timeout_request(id) {
 87             if (typeof requests[id] !== 'undefined') {
 88                 requests[id].callback({'error': 'timeout', 'result': null});
 89                 delete requests[id];
 90             }
 91         }
 92         function recv_message(e) {
 93             var pack = [], r;
 94             var ui8v = new Uint8Array(e.data);
 95 
 96             for (var i = 0; i < ui8v.length; i++) {
 97                 pack[i] = ui8v[i];
 98             }
 99             unpacker.feed(pack);
100             while ((r = unpacker.unpack()) !== undefined) {
101                 if (r[0] == 1 && typeof requests[r[1]] !== 'undefined') {
102                     clearTimeout(requests[r[1]].tid);
103                     requests[r[1]].callback({'error': r[2], 'result': r[3]});
104                     delete requests[r[1]];
105                 }
106                 if (that.callbacks && typeof that.callbacks.notify === 'function' &&
107                     r[0] == 2) {
108                     that.callbacks.notify({'method': r[1], 'params': r[2]});
109                 }
110             }
111         }
112         function recv_event(e) {
113             if (that.callbacks && typeof that.callbacks.event === 'function') {
114                 that.callbacks.event(e);
115             }
116         }
117 
118         /**
119          * do RPC async
120          * @memberOf globalScope.msgpack.rpc.client.prototype
121          * @param {Hash} rpc
122          * @param {String} rpc.method method name for RPC
123          * @param {Array} rpc.params params for RPC
124          * @param {Function} [rpc.callback]
125          * called this function when received response<br>
126          * @param {Number} [rpc.timeout]
127          * timeout time[ms], default 30000 ms
128          * @see #event:response_callback
129          */
130          that.call_async = function(rpc) {
131             msgid = (msgid == 0x0ffffffff) ? 0 : msgid + 1;
132             if (!connected || typeof requests[msgid] !== 'undefined') {
133                 setTimeout(function() {
134                                that.call_async(rpc);
135                            }, 50);
136                 return;
137             }
138             send_request(msgid, rpc);
139         };
140 
141         // initialize
142         try {
143             sock = new WebSocket(uri);   
144         } catch (x) {
145             return undefined;
146         }
147         sock.binaryType = 'arraybuffer';
148         sock.onopen = function(e) {
149             connected = true;
150             recv_event(e);
151         };
152         sock.onclose = sock.onerror = recv_event;
153         sock.onmessage = recv_message;
154         that.callbacks = callbacks;
155         return that;
156         /**
157          * fire when receive websocket event message
158          * @name globalScope.msgpack.rpc.client#event_callback
159          * @event
160          * @param {Hash} e
161          * @param {String} e.type type of event
162          * [open, close, error]
163          */
164         /**
165          * fire when receive notify message from a server
166          * @name globalScope.msgpack.rpc.client#notify_callback
167          * @event
168          * @param {Hash} e
169          * @param {String} e.method method name
170          * @param {Array} e.params arguments array
171          */
172         /**
173          * fire when receive response
174          * @name globalScope.msgpack.rpc.client#responce_callback
175          * @event
176          * @param {Hash} r
177          * @param {String} r.error
178          * error message from a server
179          * @param {Object that you define} r.result
180          * response object from a server
181          */
182     }
183 })(this);
